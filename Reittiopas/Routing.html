def routing(from, to, opts)
  @from = from
  @to = to
  @options = opts
  
  params = { :a => @from.coordinates[:kkj].to_routing_string,
             :b => @to.coordinates[:kkj].to_routing_string}.merge(@options)
  
  xml = @http.get(params)
  doc = Nokogiri::XML(xml)

  routes = []
  
  doc.search("ROUTE").each do |route|
    routes << Route.parse(route)
  end

  return routes
end
