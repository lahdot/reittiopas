def self.parse(xml)
  opts = super
  
  code = xml.get_attr_value "code"
  stop_id = xml.get_attr_value "id"
  
  name_elements = xml.search("NAME")
  
  names = {}
  name_elements.each do |e|
    lang = e.get_attr_value "lang"
    val = e.get_attr_value "val"
    
    names[lang] = val
  end
  
  opts.merge!(:code => code,
              :stop_id => stop_id,
              :names => names)
  new(opts)
end
