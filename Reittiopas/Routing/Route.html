def self.parse(xml)
  length_element = xml.search("LENGTH").first
  
  time = length_element.get_attr_value "time"
  dist = length_element.get_attr_value "dist"
    
  parts, walks, lines = [], [], []
  
  xml.elements.each do |e|
    next if e.name == "LENGTH"
    case e.name
      when "POINT"
        parts << Point.parse(e)
      when "WALK"
        walk = Walk.parse(e)
        parts << walk
        walks << walk
      when "LINE"
        line = Line.parse(e)
        parts << line
        lines << line
    end

  end

  new(:time => time,
      :distance => dist,
      :parts => parts,
      :walks => walks,
      :lines => lines)
      
end
